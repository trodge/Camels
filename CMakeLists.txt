cmake_minimum_required(VERSION 3.7)
project(Camels)
set(SRCS main.cpp game.cpp nation.cpp town.cpp business.cpp traveler.cpp ai.cpp good.cpp material.cpp settings.cpp textbox.cpp scrollbox.cpp menubutton.cpp selectbutton.cpp loadbar.cpp printer.cpp draw.cpp)
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -Wall")
set(CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake)
find_package(SDL2 REQUIRED)
find_package(Boost REQUIRED COMPONENTS system filesystem)
include_directories(${SDL2_INCLUDE_DIRS} ${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIR})
add_executable(camels ${SRCS})
target_link_libraries(camels sqlite3 ${SDL2_LIBRARIES} SDL2_ttf ${Boost_LIBRARIES})
